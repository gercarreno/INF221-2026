% ============================================================
%  handmade.tex — Sello "created with HUMAN INTELLIGENCE"
%  Requiere: TikZ  (paquete tikz + decorations.pathmorphing)
%  Uso en main.tex:  \input{handmade}   o   \include{handmade}
% ============================================================

\usepackage{tikz}
\usetikzlibrary{decorations.pathmorphing, decorations.text}

% Color rojo del sello
\definecolor{stampred}{RGB}{196, 48, 48}

% ============================================================
%  Comando reutilizable: \handmadestamp
%  Podés llamarlo donde quieras dentro de un entorno tikzpicture
%  o simplemente copiar el bloque \begin{tikzpicture}…\end
% ============================================================
\newcommand{\handmadestamp}{%
\begin{tikzpicture}[
    every path/.style={
        draw = stampred,
        line cap = round,
        line join = round,
    }
]

% ------------------------------------------------------------------
% 1. Borde exterior — forma de "squircle" dibujada a mano
%    Usamos decorate+snake para darle el temblor de mano
% ------------------------------------------------------------------
\draw[
    line width = 3.2pt,
    decorate,
    decoration={
        random steps,
        segment length = 6pt,
        amplitude      = 1.2pt
    }
]
  % Squircle aproximado con curvas de Bézier
  (-2.8,  2.4) .. controls (-2.85, 3.1) and (-2.1,  3.7) .. (-1.1,  3.75)
               .. controls ( 0.0,  3.85) and ( 1.0,  3.85) .. ( 1.1,  3.75)
               .. controls ( 2.1,  3.7)  and ( 2.85, 3.1)  .. ( 2.8,  2.4)
               .. controls ( 2.95, 1.4)  and ( 2.95,-1.4)  .. ( 2.8, -2.4)
               .. controls ( 2.85,-3.1)  and ( 2.1, -3.7)  .. ( 1.1, -3.75)
               .. controls ( 0.0, -3.85) and (-1.0, -3.85) .. (-1.1, -3.75)
               .. controls (-2.1, -3.7)  and (-2.85,-3.1)  .. (-2.8, -2.4)
               .. controls (-2.95,-1.4)  and (-2.95, 1.4)  .. (-2.8,  2.4);

% ------------------------------------------------------------------
% 2. Texto curvo superior: "created with"
%    decorations.text nos permite seguir un arco
% ------------------------------------------------------------------
\draw[
    line width = 0pt,          % el path es invisible; solo lleva el texto
    postaction = {
        decorate,
        decoration = {
            text along path,
            text = {created with},
            text align = center,
            raise = 2pt
        }
    },
    decorate,
    decoration = {
        text along path,
        text = {created with},
        text align  = center,
        raise       = 2pt
    }
]
  % Arco superior — radio ≈ 2.5, de −145° a −35°
  ({2.5*cos(215)}, {2.5*sin(215)})
  arc[start angle=215, end angle=325, radius=2.5];

% Redibujamos con estilo tipográfico manual usando \textsf inclinado
\begin{scope}[stampred]
  \path[
      postaction={
          decorate,
          decoration={
              text along path,
              text align = center,
              text={|\fontfamily{pzc}\fontsize{11}{11}\selectfont\color{stampred}|created with},
              raise = 3pt
          }
      }
  ]
  ({2.45*cos(216)},{2.45*sin(216)})
   arc[start angle=216,end angle=324,radius=2.45];
\end{scope}

% ------------------------------------------------------------------
% 3. Texto central: "HUMAN" — grande, ligeramente irregular
% ------------------------------------------------------------------
\node[
    text            = stampred,
    font            = \fontfamily{pzc}\bfseries\fontsize{28}{28}\selectfont,
    rotate          = 2,          % leve inclinación manual
    inner sep       = 0pt,
] at (0, 0.45) {HUMAN};

% ------------------------------------------------------------------
% 4. Texto inferior: "INTELLIGENCE" — más pequeño, curvo hacia abajo
% ------------------------------------------------------------------
\begin{scope}[stampred]
  \path[
      postaction={
          decorate,
          decoration={
              text along path,
              text align = center,
              text={|\fontfamily{pzc}\bfseries\fontsize{11}{11}\selectfont\color{stampred}|INTELLIGENCE},
              raise = -13pt
          }
      }
  ]
  ({2.4*cos(200)},{2.4*sin(200)})
   arc[start angle=200,end angle=340,radius=2.4];
\end{scope}

\end{tikzpicture}%
}

% ------------------------------------------------------------------
%  Render de prueba (solo cuando se compila como standalone)
% ------------------------------------------------------------------
\handmadestamp
